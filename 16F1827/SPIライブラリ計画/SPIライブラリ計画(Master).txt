
SSP1CON1 = 001ABCDE

A(クロック極性) 0:(アイドルLOW、アクティブHIGH)◎
		1:(アイドルHIGH、アクティブLOW） 

BCDE(マスターモードで分周比) 0000:(Fosc/4)
		　　　 	　　 0001:(Fosc/16)
		　　　       0010:(Fosc/64)
		             0011:(Fosc/2)

SSP1STAT = 0bFG00000H

F(サンプリング位置)　0:(入力データの中央でサンプリング)　◎
		　　 1:(入力データの末尾の縁でサンプリング)

G(クロック位相) 0:(アイドルからアクティブ)◎
		1:(アクティブからアイドル)


H(SSP1BUFの受信状態) 0:(受信未完了)
		     1:(受信完了)

SDO1SEL(SDO1の設定) 0:(RB2)
                    1:(RA6)
